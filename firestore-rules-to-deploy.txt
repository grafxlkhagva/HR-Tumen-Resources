rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper functions
    function isOwner(userId) {
      return request.auth != null && request.auth.uid == userId;
    }

    function isSignedIn() {
      return request.auth != null;
    }
    
    function isAdmin() {
      return isSignedIn() && 
             exists(/databases/$(database)/documents/employees/$(request.auth.uid)) &&
             get(/databases/$(database)/documents/employees/$(request.auth.uid)).data.get('role', 'employee') == 'admin';
    }

    // Default deny all
    match /{document=**} {
      allow read, write: if false;
    }

    // Allow everyone to read employees for context-heavy pages like Dashboard/Offboarding
    match /employees/{employeeId} {
      allow read: if isSignedIn();
      allow create: if isSignedIn();
      allow update: if isOwner(employeeId) || isAdmin();
    }
    
    // Subcollections of employees
    match /employees/{employeeId}/{subcollection}/{docId} {
       allow read, write: if isOwner(employeeId) || isAdmin();
    }

    // Explicitly allow collection group queries for offboarding_processes
    match /{path=**}/offboarding_processes/{docId} {
        allow read: if isSignedIn();
        allow write: if isAdmin();
    }

    match /employees/{employeeId}/point_profile/{docId} {
       allow read, write: if isSignedIn();
    }

    // Allow mentors/buddies to update assigned programs (e.g. mark tasks as done)
    match /employees/{employeeId}/assignedPrograms/{programId} {
      allow read: if isSignedIn();
      allow create, delete: if isAdmin();
      allow update: if isSignedIn();
    }

    // --- Points & Social System Rules ---
    match /recognition_posts/{postId} {
      allow read, create: if isSignedIn();
      allow update: if isSignedIn() && (
        isAdmin() || 
        request.auth.uid == (resource.data.fromUserId) ||
        request.resource.data.diff(resource.data).affectedKeys().hasOnly(['reactions'])
      );
      allow delete: if isAdmin() || (isSignedIn() && request.auth.uid == (resource.data.fromUserId));
    }

    match /posts/{postId} {
      allow read: if isSignedIn();
      allow create, delete: if isAdmin();
      allow update: if isAdmin() || 
                    (isSignedIn() && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['reactions']));
    }

    match /points_config/{docId} {
      allow read: if isSignedIn();
      allow write: if isAdmin();
    }

    match /point_transactions/{txId} {
      allow read: if isSignedIn() && (isAdmin() || request.auth.uid == resource.data.userId);
      allow create: if isSignedIn();
    }

    match /rewards/{rewardId} {
      allow read: if isSignedIn();
      allow write: if isAdmin();
    }

    match /redemption_requests/{requestId} {
      allow read: if isSignedIn() && (isAdmin() || request.auth.uid == resource.data.userId);
      allow create: if isSignedIn();
      allow update: if isAdmin();
    }

    match /budget_point_requests/{requestId} {
      allow create: if isSignedIn();
      allow read, update, delete: if isSignedIn() && (isAdmin() || request.auth.uid == resource.data.fromUserId);
    }



    // Vacation requests rules
    match /employees/{employeeId}/vacationRequests/{requestId} {
      allow read: if isSignedIn();
      allow create: if isOwner(employeeId);
      allow update: if isOwner(employeeId) || isAdmin() || request.auth.uid == resource.data.approverId;
      allow delete: if isAdmin() || isOwner(employeeId);
    }
    
    // Company profile and branding settings
    match /company/{allPaths=**} {
      allow read: if isSignedIn() || true;
      allow write: if isAdmin();
    }

    // Company-wide policies can be read by any signed-in user, but only managed by admins.
    match /companyPolicies/{policyId} {
        allow read: if isSignedIn();
        allow write: if isAdmin();
    }

    // Reference data (collections ending with 'Types', 'Levels', 'Statuses', etc.) is readable by any signed-in user
    // Writable only by admins
    match /{refCollection}/{docId} {
       allow read: if isSignedIn();
       allow write: if isAdmin();
    }

    // All users can read departments, positions, and company info
    match /departments/{departmentId} {
      allow read: if isSignedIn();
      allow write: if isAdmin();
    }
    
    match /positions/{positionId} {
      allow read: if isSignedIn();
      allow write: if isAdmin();
    }


    // Onboarding program templates are managed by admins
    match /onboardingPrograms/{programId}/{document=**} {
       allow read: if isSignedIn();
       allow write: if isAdmin();
    }

    // Onboarding-related collections are managed by admins
    match /newHires/{hireId} {
        allow read, write: if isAdmin();
    }

    // Attendance records can be created by the user, but likely only read by admins or managers
    match /attendance/{attendanceId} {
      allow create: if isOwner(request.resource.data.employeeId);
      allow read, update: if isAdmin();
    }

    // The following comment is a special instruction for the backend to create an index.
    // Please do not modify or remove it.
    // FIRESTORE_INDEX: COLLECTION_GROUP(timeOffRequests) ASC(status)

    // Collection Group rules
    match /{path=**}/assignedPrograms/{docId} {
        allow read: if isSignedIn();
        allow write: if isAdmin();
    }

    match /{path=**}/timeOffRequests/{docId} {
        allow read: if isSignedIn();
        allow write: if isAdmin();
    }

    match /{path=**}/vacationRequests/{docId} {
        allow read: if isSignedIn();
        allow write: if isAdmin() || (isSignedIn() && request.auth.uid == resource.data.approverId);
    }
    
    
    // --- Employment Relations Module Rules ---
    
    // Document Types - Admin only
    match /er_document_types/{typeId} {
      allow read: if isSignedIn();
      allow write: if isAdmin();
    }
    
    // Templates - Admin only for write, all signed-in users can read active templates
    match /er_templates/{templateId} {
      allow read: if isSignedIn();
      allow write: if isAdmin();
    }
    
    // Workflows - Admin only
    match /er_workflows/{workflowId} {
      allow read: if isSignedIn();
      allow write: if isAdmin();
    }
    
    // Documents - Complex rules based on status and role
    match /er_documents/{documentId} {
      // Read access:
      // - Creator can always read their own documents
      // - Reviewers can read documents they need to review
      // - Admins can read all
      allow read: if isSignedIn() && (
        isAdmin() ||
        request.auth.uid == resource.data.creatorId ||
        request.auth.uid == resource.data.employeeId ||
        request.auth.uid in resource.data.get('reviewers', []) ||
        (exists(/databases/$(database)/documents/employees/$(request.auth.uid)) &&
         get(/databases/$(database)/documents/employees/$(request.auth.uid)).data.get('positionId', '') in resource.data.get('reviewers', []))
      );
      
      // Create: Any signed-in user can create documents
      allow create: if isSignedIn();
      
      // Update access:
      // - Creator can update their own DRAFT documents
      // - Reviewers can update approvalStatus when document is IN_REVIEW
      // - Admins can update anything
      allow update: if isSignedIn() && (
        isAdmin() ||
        // Creator updating their own draft
        (request.auth.uid == resource.data.creatorId && resource.data.status == 'DRAFT') ||
        // Reviewer updating approval status
        (resource.data.status == 'IN_REVIEW' && 
         (request.auth.uid in resource.data.get('reviewers', []) ||
          (exists(/databases/$(database)/documents/employees/$(request.auth.uid)) &&
           get(/databases/$(database)/documents/employees/$(request.auth.uid)).data.get('positionId', '') in resource.data.get('reviewers', []))) &&
         // Only allow updating specific fields
         request.resource.data.diff(resource.data).affectedKeys().hasOnly(['approvalStatus', 'status', 'updatedAt', 'rejectionReason']))
      );
      
      // Delete: Only creator can delete their own drafts, or admins
      allow delete: if isSignedIn() && (
        isAdmin() ||
        (request.auth.uid == resource.data.creatorId && resource.data.status == 'DRAFT')
      );
    }
    
    // --- Recruitment Module Rules ---
    
    match /vacancies/{vacancyId} {
        allow read: if true; // Public read for job board (or use a cloud function for public access to be safer)
        allow write: if isAdmin();
    }
    
    match /candidates/{candidateId} {
        allow create: if true; // Public application submission
        allow read, update, delete: if isAdmin();
    }
    
    match /applications/{applicationId} {
        allow create: if true; // Public application submission
        allow read, update, delete: if isAdmin();
    }
    
    match /interviews/{interviewId} {
        allow read: if isSignedIn();
        allow write: if isAdmin();
    }

  }
}
